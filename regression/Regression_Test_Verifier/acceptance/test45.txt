-- test disjunction with initialized boolean variable declaration with relational value + propositional value (no error)
i : INTEGER = 1 + 3
j : INTEGER = i * 2
m : INTEGER = 3 * 4 + 5
p : BOOLEAN = not i > 0
q : BOOLEAN = i > j or m = i * j
verify p or q
r : BOOLEAN = j >= m and not p
s : BOOLEAN = j < 20 or m = 17
verify not r or s or q
s1 : BOOLEAN = j = i => s
r0 : BOOLEAN = i = j <=> s1
k : BOOLEAN = m < i and not r
verify not r0 or s1 or not k or r
