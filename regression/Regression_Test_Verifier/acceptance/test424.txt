-- test program verification
-- if-else statement (Not discharged)
x: INTEGER
y: INTEGER
bigger: INTEGER
smaller: INTEGER

bigger_or_smaller()
require 
	positive: x > 0 and y > 0
do
if x > y then
	bigger := x;
	smaller := y;
else
	bigger := x;
	smaller := y;
end
ensure
	ensured: bigger >= smaller <=> true and false
end

verify bigger_or_smaller
