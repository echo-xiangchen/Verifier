j : INTEGER = i * 2
m : INTEGER = 3 * 4 + 5
p : BOOLEAN = i > 0
q : BOOLEAN = j >= i
m : INTEGER = i / j - 4
verify not p or q => q
r : BOOLEAN = j >= m
s : BOOLEAN = j < 20
verify r => s and (not p <=> q)
s : INTEGER = m + i
verify not s <=> (p and (q => r or q))
