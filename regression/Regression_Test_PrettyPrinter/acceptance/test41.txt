-- test disjunction with initialized boolean variable declaration with propositional value (no error)
p : BOOLEAN = true
q : BOOLEAN = not p and p
verify q or not p
r : BOOLEAN = p and q or not p
t1 : BOOLEAN = p or q
verify r or t1 or not p or q
verify t1 or r
t2 : BOOLEAN = t1 => q
t3 : BOOLEAN = p <=> t2
verify t3 or t2 or not t1
