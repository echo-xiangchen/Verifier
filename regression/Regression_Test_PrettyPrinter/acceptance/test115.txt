-- test propositional logic and greater than  with initialized boolean and int variable declaration with constant value(no error)
i : INTEGER = 5
j : INTEGER = 1
m : INTEGER = 35
p : BOOLEAN = true
q : BOOLEAN = false
verify p <=> q and i > j or not m > 0
r : BOOLEAN = true
s : BOOLEAN = false
verify not r and (s or q) <=> p => m > i * j and not i > m
s1 : BOOLEAN = true
r0 : BOOLEAN = false
k : BOOLEAN = false
verify not r0 <=> (s1 => (not k and r)) or i > j - m
