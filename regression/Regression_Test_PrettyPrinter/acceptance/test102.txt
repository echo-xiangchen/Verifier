-- test propositional logic with initialized boolean variable declaration with propositional value (with error)
p : BOOLEAN = true
q : BOOLEAN = not p or p
verify q and not p => not q
p : BOOLEAN = not q
r : BOOLEAN = p and q
t1 : BOOLEAN = p or q
verify r and t1 <=> not p or q
verify t1 => (not r and (p and q))
t2 : BOOLEAN = t1 => p
t3 : BOOLEAN = p <=> t2
t3 : BOOLEAN = q => r and t1
verify t3 and t2 => not (t1 or (r => q))
