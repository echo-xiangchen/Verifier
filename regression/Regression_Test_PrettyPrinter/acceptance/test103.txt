-- test propositional logic with initialized boolean variable declaration with relational value (no error)
i : INTEGER = 1 + 3
j : INTEGER = 2
m : INTEGER = 3 * 4 + 5
p : BOOLEAN = i > 0
q : BOOLEAN = j >= i
verify not p or q => q
r : BOOLEAN = j >= m
s : BOOLEAN = j < 20
verify r => s and (not p <=> q)
verify not s <=> (p and (q => r or q))
