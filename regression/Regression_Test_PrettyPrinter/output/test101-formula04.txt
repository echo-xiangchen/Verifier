(declare-const p Bool)
(assert (= p true))
(declare-const q Bool)
(assert (= q (or (not p) p)))
(declare-const r Bool)
(assert (= r (and p q)))
(declare-const t1 Bool)
(assert (= t1 (or p q)))
(declare-const t2 Bool)
(assert (= t2 (=> t1 q)))
(declare-const t3 Bool)
(assert (= t3 (= p t2)))
(assert (not (=> (and t3 t2) (not (or t1 (=> r q))))))
