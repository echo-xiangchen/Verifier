(declare-const p Bool)
(assert (= p true))
(declare-const q Bool)
(assert (= q (not p)))
(declare-const r Bool)
(assert (= r (=> (not p) (and q p))))
(declare-const s Bool)
(assert (= s (= (and r p) (and q p))))
(declare-const i Int)
(assert (= i (- (* 5 6) 9)))
(declare-const j Int)
(assert (= j (+ 1 9)))
(declare-const m Int)
(assert (= m (- 35 (* 12 2))))
(declare-const n Int)
(assert (= n (- (+ i (* m j)) 25)))
(assert (not (= (and (not r) (or s q)) (=> (and (or p (= m (* i n))) (not (>= i m))) (> n (+ (* m i) j))))))
